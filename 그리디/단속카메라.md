```python
def solution(routes):
    answer = 0
    cnt =0
    tmp =[0]*len(routes)
    
    for i in range(len(routes)-1):
        for j in range(i+1,len(routes)):
            if tmp[i]!=1:
                if tmp[j]!=1 and routes[i][0]<=routes[j][0] and routes[i][1] <= routes[j][1]:
                    tmp[j]=1
                    answer +=1
            else:
                break
        tmp[i]=1
        if 0 not in tmp:
            return answer
    return answer
```
